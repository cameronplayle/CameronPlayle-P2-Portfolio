  step_normalize(all_predictors())

# Define a logistic regression model

  set_engine("glm") %>%
  set_mode("classification")

# Create a workflow
logistic_workflow <- workflow() %>%
  add_recipe(recipe) %>%
  add_model(logistic_model)

  fit(data = train_data)
